DESPUÉS:

  with qgisapp(sysexit=False) as qgs_app:
      ...  ✓ CORRECTO


[2] CARGA DE ALGORITMOS (processing_chain_poc.py)
───────────────────────────────────────────────

ANTES:
  registry = QgsApplication.processingRegistry()
  algorithms = registry.algorithms()  # Podría estar vacía
  for algo in algorithms:
      combo.addItem(...)

DESPUÉS:
  registry = QgsApplication.processingRegistry()
  
  # Inicializar si es necesario
  if len(list(registry.algorithms())) == 0:
      from qgis.analysis import QgsNativeAlgorithms
      registry.addProvider(QgsNativeAlgorithms())
  
  algorithms = list(registry.algorithms())
  
  # Limpiar y recargar
  combo.clear()
  for algo in algorithms:
      combo.addItem(...)


[3] VISUALIZACIÓN DEL PROYECTO (qgs_viewer_poc.py)
──────────────────────────────────────────────────

ANTES:
  project.read(file_path)
  canvas.setProject(project)
  canvas.zoomToFullExtent()  # Podría no funcionar sin capas

DESPUÉS:
  project.read(file_path)
  canvas.setProject(project)
  
  # Configurar capas explícitamente
  layers = list(project.mapLayers().values())
  canvas.setLayers(layers)
  
  # Asegurar visualización
  canvas.zoomToFullExtent()
  canvas.refresh()


[4] INTERFAZ DE USUARIO (processing_chain_poc.py)
─────────────────────────────────────────────────

ANTES:
  layout.addWidget(self.combo_algorithms)

DESPUÉS:
  algo_combo_layout = QHBoxLayout()
  algo_combo_layout.addWidget(self.combo_algorithms, 1)
  
  btn_reload = QPushButton("🔄")
  btn_reload.setMaximumWidth(40)
  btn_reload.setToolTip("Recargar algoritmos")
  btn_reload.clicked.connect(self.load_algorithms)
  algo_combo_layout.addWidget(btn_reload)
  
  layout.addLayout(algo_combo_layout)

POSIBLES ERRORES Y SOLUCIONES:
═══════════════════════════════

❌ "No se encontraron algoritmos"
   → Ejecuta: python debug_poc.py
   → Pulsa botón 🔄 en viewer

❌ "Proyecto cargado pero sin capas"
   → Verifica que el .qgs existe
   → Prueba con otro proyecto

❌ AttributeError: 'QgsApplication' object has no attribute 'initQgisApplication'
   → YA DEBE ESTAR SOLUCIONADO
   → Si aparece de nuevo, asegúrate que actualizaste qgs_viewer_poc.py

❌ Canvas vacío después de cargar proyecto
   → Comprueba que las capas se cargaron (verás número en status bar)
   → Intenta hacer zoom manualmente en canvas
